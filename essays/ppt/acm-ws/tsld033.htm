More list operations

    >>> a = range(5) # [0,1,2,3,4]

    >>> a.append(5) # [0,1,2,3,4,5]

    >>> a.pop() # [0,1,2,3,4]

    5

    >>> a.insert(0, 5.5) # [5.5,0,1,2,3,4]

    >>> a.pop(0) # [0,1,2,3,4]

    5.5

    >>> a.reverse() # [4,3,2,1,0]

    >>> a.sort() # [0,1,2,3,4]
